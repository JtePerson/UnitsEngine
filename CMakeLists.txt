cmake_minimum_required(VERSION 3.20)
project(UnitsRepo LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED true)
set(CMAKE_CONFIGURATION_TYPES "Debug;Release" CACHE STRING "" FORCE)
set(CMAKE_EXPORT_COMPILE_COMMANDS true)

option(BUILD_ENGINE  "Build UnitsEngine" true)
option(BUILD_TESTS   "Build UnitTests"   true)
option(BUILD_EDITOR  "Build UnitsEditor" true)
option(BUILD_GAME "Build UnitsGame"     true)

if(BUILD_ENGINE)
	add_subdirectory(${PROJECT_SOURCE_DIR}/UnitsEngine)
endif()
if(BUILD_TESTS)
	add_subdirectory(${PROJECT_SOURCE_DIR}/UnitTests)
endif()
if(BUILD_EDITOR)
	add_subdirectory(${PROJECT_SOURCE_DIR}/UnitsEditor)
endif()
if(BUILD_GAME)
	add_subdirectory(${PROJECT_SOURCE_DIR}/UnitsGame)
endif()
